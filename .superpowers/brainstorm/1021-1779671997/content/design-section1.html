🔧 后端权限检查

# permissions.py — 统一权限检查函数

def check_permission(role: str, action: str) -> bool:
    """检查角色是否有某项权限"""
    allowed_roles = PERMISSION_MATRIX.get(action)
    if not allowed_roles:
        return False
    role_idx = ROLE_ORDER.index(role) if role in ROLE_ORDER else -1
    return role_idx >= 0 and allowed_roles[role_idx]

def require_project_permission(action: str):
    """FastAPI 依赖：检查当前用户在项目中的角色权限"""
    async def _check(
        project_id: UUID,
        user: CurrentUser,
        db: AsyncSession = Depends(get_db),
    ):
        member = await _get_member_role(db, project_id, user.id)
        if not member and not user.is_admin:
            raise HTTPException(403, "不是项目成员")
        role = "manager" if user.is_admin else member.role
        if not check_permission(role, action):
            raise HTTPException(403, f"角色 '{role}' 无权执行 '{action}'")
        return role
    return _check

🎨 前端权限消费

// hooks/useProjectPermissions.ts

// API 返回用户在项目中的角色 + 权限列表
interface ProjectMembership {
  role: string;
  permissions: string[];  // ["outline:edit", "chapter:write_own", ...]
}

function useProjectPermissions(projectId: string) {
  const { data } = useQuery({
    queryKey: ["project-membership", projectId],
    queryFn: () => fetchProjectMembership(projectId),
  });

  const can = (action: string) =>
    data?.permissions.includes(action) ?? false;

  return { role: data?.role, can, permissions: data?.permissions ?? [] };
}

// 组件内使用：
const { can } = useProjectPermissions(projectId);
{can("member:add") && <Button>添加成员</Button>}

✅ 设计要点

  • 单一数据源：权限只在 permissions.py 定义一次，前端通过 API 获取
  • admin 超级权限：系统管理员自动拥有 manager 级权限
  • 项目创建者：自动成为 manager 角色
  • 优雅降级：前端隐藏无权限按钮，后端拒绝无权限请求，双重保护
  • 新增 API：GET /projects/{id}/my-permissions 返回当前用户的权限列表
